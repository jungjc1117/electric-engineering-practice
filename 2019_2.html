$ Z{HM} = 10 = 2 [%], Z{ML} = 67 = 13.4 [%], Z_{HL} = 78 = 15.6 [%] $

\[ \%Z*H = \frac{1}{2}(Z*{HM} + Z*{HL} - Z*{ML}) = \frac{1}{2}(2 + 15.6 - 13.4) = 2.1 [\%] \]

\[ \%Z*M = \frac{1}{2}(Z*{HM} + Z*{ML} - Z*{HL}) = \frac{1}{2}(2 + 13.4 - 15.6) = -0.1 [\%] \]
